3.Create a promise function accepting a argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject state and catch the error and print Promise Rejected 

        let printNumber =(time, num) =>{
             return new Promise ((resolve, reject)=>{
                   if (num){
                       setTimeout(()=>{
                           resolve(num());
                       },time)
                   }
                   else{
                       reject(console.log("failed"))
                   }
               })
             }
             printNumber(0, () =>console.log("Passed")).then(()=>{
               return printNumber(1000,() =>{console.log("1")})
             })
